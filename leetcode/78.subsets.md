# 78. Subsets


https://leetcode.com/problems/subsets/

## My solution

```js
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function(nums) {
    const result = []

    const dfs = (index, path) => {
        result.push(path)

        for (let i = index; i < nums.length; i += 1) {
            dfs(i + 1, path.concat(nums[i]))
        }
    }

    dfs(0, [])

    return result
};
```
